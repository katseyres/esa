PROGRAMME verificationMotDePasse

motDePasse, alphaLowerPattern, alphaUpperPattern, digitPattern, specialCharPattern : CHAINE

DEBUT
    LIRE(motDePasse)

    alphaLowerPattern   <- "abc(...)xyz"
    alphaUpperPattern   <- "ABC(...)XYZ"
    digitPattern        <- "012(...)789"
    specialCharPattern  <- # tout ce qui n'est pas dans les 3 variables précédentes

    SI longueur(motDePasse) < 10 ALORS
        RETOURNE Faux
    SINON SI motDePasse N'A PAS D'ELEMENT DANS alphaLowerPattern ALORS
        RETOURNE Faux
    SINON SI motDePasse N'A PAS D'ELEMENT DANS alphaUpperPattern ALORS
        RETOURNE Faux
    SINON SI motDePasse N'A PAS D'ELEMENT DANS digitPattern ALORS
        RETOURNE Faux
    SINON SI motDePasse N'A PAS D'ELEMENT DANS specialCharPattern ALORS
        RETOURNE Faux
    FIN

    RETOURNE Vrai
FIN
